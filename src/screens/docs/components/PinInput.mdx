export function WithSubmit() {
  return (
    <PinInput.Root>
      <PinInput.Label>Verify code (5 digits)</PinInput.Label>
      <PinInput.Control>
        {Array.from({ length: 5 }).map((_, i) => (
          <PinInput.Input key={i} index={i} />
        ))}
      </PinInput.Control>
      <PinInput.Context>
        {(api) => {
          const filled = api.value.filter(Boolean).length;
          const total = api.value.length;
          return (
            <div className="mt-1 flex items-center justify-between gap-3">
              <span className="text-fg-muted text-[11px]">
                {filled} / {total} digits
              </span>
              <Button size="sm" disabled={!api.complete}>
                Verify
              </Button>
            </div>
          );
        }}
      </PinInput.Context>
    </PinInput.Root>
  );
}

# PinInput

OTP / PIN / 認証コード入力用の桁ごと box。 paste で全桁に分配、 入力毎に次の box へ自動 focus、 全桁埋まると onValueComplete が発火。 otp 属性で iOS / Android の SMS 自動入力にも対応。

## Anatomy

Root の中に Label / Control / (Context 任意)。 Control の中に Input を桁数分。

<ul className="flex flex-col gap-2 rounded-md border border-border p-4">

  ))}
</ul>

## With submit gate

Context render-prop で api.complete / api.value を取って、 全桁埋まるまで submit を disable + 進捗表示。

<Example
  code={`<PinInput.Root>
  <PinInput.Label>Verify code (5 digits)</PinInput.Label>
  <PinInput.Control>
    {Array.from({ length: 5 }).map((_, i) => <PinInput.Input key={i} index={i} />)}
  </PinInput.Control>
  <PinInput.Context>
    {(api) => (
      <div className="flex justify-between">
        <span>{api.value.filter(Boolean).length} / {api.value.length} digits</span>
        <Button disabled={!api.complete}>Verify</Button>
      </div>
    )}
  </PinInput.Context>
</PinInput.Root>`}
>
  <WithSubmit />
</Example>
